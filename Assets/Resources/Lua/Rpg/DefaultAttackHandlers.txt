function Hit(attacker, defender)
    local outcome = {
        hit=true,
        description='hit',
        damageToTarget=math.max(math.floor(attacker.damage.Total - defender.defense.Total), configuration.minimum_attack_damage),
        attacker=attacker
    }
    local onHitTriggers = attacker.OnEventTriggers['IsAttacking']
    return outcome
end

function Miss(attacker, defender)
    local outcome = {
        hit=false,
        description='miss',
        attacker=attacker
    }
    return outcome
end

function CriticalHit(attacker, defender)
    local outcome = {
        hit=true,
        description='critical hit',
        damageToTarget=math.floor(math.max((attacker.damage.Total - defender.defense.Total), 1) * (1 + (attacker.criticalHitDamageMultiplier.Total / 100))),
        attacker=attacker
    }
    return outcome
end

AttackHandlers = {
    hit = Hit,
    miss = Miss,
    criticalHit = CriticalHit
}
